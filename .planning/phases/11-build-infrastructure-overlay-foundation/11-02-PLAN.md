       match apply_mica(&window, Some(true)) {

               // RGBA: (18, 18, 18, 125) matches macOS HudWindow darkness
               apply_acrylic(&window, Some((18, 18, 18, 125)))

               eprintln!("[setup] Acrylic vibrancy applied (Windows 10)");
           }
       }
   }

   #[cfg(target_os = "windows")]

       use raw_window_handle::{HasWindowHandle, RawWindowHandle};
       use windows_sys::Win32::UI::WindowsAndMessaging::*;

       if let Ok(handle) = window.window_handle() {
           if let RawWindowHandle::Win32(win32) = handle.as_raw() {
               let hwnd = win32.hwnd.get() as isize;
               unsafe {
                   let ex_style = GetWindowLongPtrW(hwnd, GWL_EXSTYLE);
                   let new_style = (ex_style | WS_EX_TOOLWINDOW as isize)
                       & !(WS_EX_APPWINDOW as isize);
                   SetWindowLongPtrW(hwnd, GWL_EXSTYLE, new_style);

               eprintln!("[setup] WS_EX_TOOLWINDOW applied (hwnd={})", hwnd);
           }
       }
   }

Refactor window.rs to have fully functional Windows paths for show_overlay and hide_overlay.

**show_overlay -- Windows path:**
The macOS path uses `app.get_webview_panel("main")` and `panel.show_and_make_key()`. The Windows path should:
1. Get the window via `app.get_webview_window("main")`
2. Call `position_overlay(&app)?` (same as macOS -- this is cross-platform)
3. Call `window.show().map_err(|e| e.to_string())?`
4. Call `window.set_focus().map_err(|e| e.to_string())?` (Windows equivalent of show_and_make_key)
5. Update visibility state (same logic as macOS)
6. Emit "overlay-shown" event (same as macOS)

Structure as:
```rust
#[tauri::command]
pub fn show_overlay(app: AppHandle) -> Result<(), String> {
    position_overlay(&app)?;

    #[cfg(target_os = "macos")]
    {
        let panel = app
            .get_webview_panel("main")
            .map_err(|e| format!("Panel 'main' not found: {:?}", e))?;
        panel.show_and_make_key();
    }

    #[cfg(not(target_os = "macos"))]
    {
        let window = app
            .get_webview_window("main")
            .ok_or_else(|| "Window 'main' not found".to_string())?;
        window.show().map_err(|e| e.to_string())?;
        window.set_focus().map_err(|e| e.to_string())?;
    }

    // Update visibility state (cross-platform)
    if let Some(state) = app.try_state::<AppState>() {
        if let Ok(mut visible) = state.overlay_visible.lock() {
            *visible = true;
        }
    }
    let _ = app.emit("overlay-shown", ());
    Ok(())
}
```

**hide_overlay -- Windows path:**
```rust
#[tauri::command]
pub fn hide_overlay(app: AppHandle) -> Result<(), String> {
    #[cfg(target_os = "macos")]
    {
        let panel = app
            .get_webview_panel("main")
            .map_err(|e| format!("Panel 'main' not found: {:?}", e))?;
        panel.hide();
    }

    #[cfg(not(target_os = "macos"))]
    {
        let window = app
            .get_webview_window("main")
            .ok_or_else(|| "Window 'main' not found".to_string())?;
        window.hide().map_err(|e| e.to_string())?;
    }

    // Update visibility state (cross-platform)
    if let Some(state) = app.try_state::<AppState>() {
        if let Ok(mut visible) = state.overlay_visible.lock() {
            *visible = false;
        }
    }
    Ok(())
}
```
